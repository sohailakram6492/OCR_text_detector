LEFT OUTER JOIN LOV_FT_TXN_TYPE_CONDITION TP ON A.TRANSACTION_TYPE = TP.ID AND A.LEAD_COMPANY = TP.LEAD_COMPANY

LEFT OUTER JOIN PK_H_IMD_LIST P ON T.ACQ_IMD = P.ID
